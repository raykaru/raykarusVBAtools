
Sub SwitchClear()

    ActiveSheet.Cells(1, 1).Value = ""
    

End Sub

Sub SwitchUpper()

    Dim tmp As String
    
    tmp = ActiveSheet.Cells(1, 1).Value
    ActiveSheet.Cells(1, 1).Value = StrConv(tmp, vbUpperCase)
    

End Sub

End Sub

Sub SwitchWhiteSpace()

    Dim tmp As String
    
    tmp = ActiveSheet.Cells(1, 1).Value
    tmp = Trim(tmp)
    tmp = Replace(tmp, vbTab, "")           'タブ削除
    tmp = Replace(tmp, vbCr, "")            '改段落削除
    tmp = Replace(tmp, vbVerticalTab, "")   '改行削除
    ActiveSheet.Cells(1, 1).Value = tmp
    

End Sub
